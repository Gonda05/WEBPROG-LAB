/krs_system                # Root folder proyek
│── /assets                # Folder untuk file statis (CSS, JS, Gambar)
│   ├── /css               # File CSS
│   ├── /img               # Gambar/icon
│   ├── /js                # File JavaScript
│── /config                # Konfigurasi database dan pengaturan lainnya
│   ├── config.php         # Konfigurasi umum aplikasi (base_url, session, dll.)
│   ├── database.php       # Koneksi ke MySQL
│── /controllers           # Folder untuk logika aplikasi (MVC Controller)
│   ├── AuthController.php  # Proses login/logout
│   ├── DosenController.php # CRUD Dosen
│   ├── KrsController.php   # CRUD Transaksi KRS
│   ├── MahasiswaController.php # CRUD Mahasiswa
│   ├── MatkulController.php # CRUD Mata Kuliah
│── /helpers               # Fungsi-fungsi umum
│   ├── helper.php         # Fungsi umum (validasi, redirect, dll.)
│── /models                # Folder untuk mengakses database (MVC Model)
│   ├── Dosen.php          # Model untuk dosen
│   ├── Krs.php            # Model untuk transaksi KRS
│   ├── Mahasiswa.php      # Model untuk mahasiswa
│   ├── Matkul.php         # Model untuk mata kuliah
│   ├── User.php           # Model untuk user (Admin, Dosen, Mahasiswa)
│── /public                # Folder untuk akses utama (entry point)
│   ├── index.php          # Halaman utama (redirect ke login jika belum login)
│   ├── login.php          # Halaman login
│   ├── logout.php         # Proses logout
│── /views                 # Folder tampilan (HTML + PHP)
│   ├── /layouts           # Template layout
│   │   ├── footer.php     # Footer template
│   │   ├── header.php     # Header template
│   │   ├── sidebar.php    # Sidebar/navigation
│   ├── admin_dashboard.php # Dashboard admin
│   ├── dosen_dashboard.php # Dashboard dosen
│   ├── home.php           # Dashboard setelah login
│   ├── mahasiswa_dashboard.php # Dashboard mahasiswa
│── .htaccess              # Konfigurasi Apache (opsional)
│── README.md              # Dokumentasi proyek
